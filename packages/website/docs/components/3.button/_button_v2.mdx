# Button Component

The `glu-button` component is a highly customizable button element that supports various appearances, sizes, and types. It functions either as a standard button or as a link when an `href` is provided.

## Basic Usage

### Example 1: Simple Button

Use the button as a clickable element.

<Tabs>
  <TabItem value="html" label="HTML">
    <Sandpack
      template="vanilla"
      files={{
        '/index.html': `<html>
          <body>
            <glu-button>Click Me</glu-button>
          </body>
        </html>`,
      }}
      options={{ showNavigator: true }}
    />
  </TabItem>
  <TabItem value="jsx" label="React">
    <Sandpack
      template="react"
      files={{
        '/App.js': `import { GluButton } from '@gelato-ui/react';

        function App() {
          return (
            <GluButton>Click Me</GluButton>
          );
        }

        export default App;`,
      }}
      options={{ showNavigator: true }}
    />
  </TabItem>
  <TabItem value="vue" label="Vue">
    <Sandpack
      template="vue"
      files={{
        '/App.vue': `<template>

        </template>

        <script setup>

        })
        export class AppComponent {}`,
      }}
      options={{ showNavigator: true }}
    />
  </TabItem>
</Tabs>

## Button in a Form

Use the button as a submission control within a form.

<Tabs>
  <TabItem value="html" label="HTML">
    <Sandpack
      template="vanilla"
      files={{
        '/index.html': `<html>
  <body>
    <form onsubmit="alert('Form submitted'); return false;">
      <glu-button type="submit">Submit Form</glu-button>
    </form>
  </body>
</html>`,
      }}
      options={{ showNavigator: true }}
    />
  </TabItem>
  <TabItem value="jsx" label="React">
    <Sandpack
      template="react"
      files={{
        '/App.js': `import { GluButton } from '@gelato-ui/react';

function FormExample() {
  const handleSubmit = (e) => {
    e.preventDefault();
    alert('Form submitted');
  };

  return (
    <form onSubmit={handleSubmit}>
      <GluButton type="submit">Submit Form</GluButton>
    </form>
  );
}

export default FormExample;`,
      }}
      options={{ showNavigator: true }}
    />
  </TabItem>
  <TabItem value="vue" label="Vue">
    <Sandpack
      template="vue"
      files={{
        '/App.vue': `<template>
  <form @submit.prevent="() => alert('Form submitted')">
    <glu-button type="submit">Submit Form</glu-button>
  </form>
</template>

<script setup>
import { GluButton } from '@gelato-ui/vue';
</script>`,
      }}
      options={{ showNavigator: true }}
    />
  </TabItem>
  <TabItem value="angular" label="Angular">
    <Sandpack
      template="angular"
      files={{
        '/src/app/app.component.html': `<form (ngSubmit)="onSubmit()">
  <glu-button type="submit">Submit Form</glu-button>
</form>`,
        '/src/app/app.component.ts': `import { Component } from '@angular/core';
import { GluButton } from '@gelato-ui/angular';

@Component({
  standalone: true,
  imports: [GluButton],
  templateUrl: './app.component.html'
})
export class AppComponent {
  onSubmit() {
    alert('Form submitted');
  }
}`,
      }}
      options={{ showNavigator: true }}
    />
  </TabItem>
</Tabs>
