### Button Slots

This example demonstrates how to use the start, end, and icon-only slots to customize the button's content.

#### Example Usage

<Tabs>
  <TabItem value="html" label="HTML">

    ```html live
      <div style={{ display: 'flex', flexDirection: 'row', gap: '1rem', justifyContent: 'center', alignItems: 'center' }}>
        <glu-button>
          <span slot="start">👈</span>
          Click Me
          <span slot="end">👉</span>
          <span slot="icon-only">⭐</span>
        </glu-button>
      </div>
    ```

  </TabItem>
  <TabItem value="jsx" label="React">

    ```jsx live

          <span slot="end">👉</span>
          <span slot="icon-only">⭐</span>
        </GluButton>
      </div>
    ```

  </TabItem>
  <TabItem value="vue" label="Vue">

    ```html
      <div style={{ display: 'flex', flexDirection: 'row', gap: '1rem', justifyContent: 'center', alignItems: 'center' }}>
        <glu-button>
          <template #start>👈</template>
          Click Me
          <template #end>👉</template>
          <template #icon-only>⭐</template>
        </glu-button>
      </div>

      <script setup>
        import { GluButton } from '@gelato-ui/vue';
      </script>
    ```

  </TabItem>
